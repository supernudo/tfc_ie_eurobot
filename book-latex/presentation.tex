\begin{frame}{Velocidad máxima}
\alt<1>{
		\begin{columns}
		\column{.5\textwidth}
		\includegraphics[width=\textwidth]{dimensionamiento_motores}

		\column{.5\textwidth}
		Tensión del motor:
		\begin{equation*}
		U = \Big(\frac{r\,m\,D}{4K\,Rd\,K_m}\Big)\,a + \Big(\frac{2K_m\,K}{D}\Big)\,v
		\end{equation*}

		Equivale a una recta:
		\begin{equation*}
		v = v_o - \frac{v_o}{a_o}\,a
		\end{equation*}
		Donde
		\begin{equation*}
		v_o = \frac{\,U\,D}{2K\,K_m}, \,\,\,\,\,\, a_o = \frac{4K\,U\,Rd\,K_m}{r\,D\,m}
		\end{equation*}

		\end{columns}
}{}

\alt<2>{

		v_o &= \frac{\,U\,D}{2K\,K_m}\\ 
		a_o &= \frac{4K\,U\,Rd\,K_m}{r\,D\,m}
		\end{empheq}
